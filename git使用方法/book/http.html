2.建立函数，按行读出内容

def getLine(sock):

    b1 = sock.recv(1)
    b2 = 0
    data = b''

    while  b2 != b'\r' and b1 != b'\n' :
        b2 = b1
        b1 = sock.recv(1)

        data += bytes(b2)

    return data.strip(b'\r')

3.建立函数，一次性读出所有请求头信息

def getAllLine(sock):

    data = b''
    dataList = list()
    data = b''

    while True:
        data = getLine(sock)
        if data:
            dataList.append(data)
        else:
            return dataList

WebSever

  • 启动使用start
  • 初始化需要用户输入IP和port，如果没有需要给出默认值
  • 服务器需要为每个请求单独处理（多线程）
  • 每传入一个socket，需要单独起一个线程处理
import socket
import threading

from sockhandler import SocketHandler


class WebServer():


    def __init__(self, ip='127.0.0.1', port=7853):
        self.ip = ip
        self.port = port

        self.sock = socket.socket( socket.AF_INET, socket.SOCK_STREAM)
        self.sock.bind((self.ip, self.port))
        self.sock.listen(1)
        print("WebServer is started............................")

    def start(self):
        while True:
            skt, addr = self.sock.accept()

            if skt:
                print("Received a socket {0} from {1} ................. ".format(skt.getpeername(), addr))
                sockHandler = SocketHandler(skt)
                thr = threading.Thread(target=sockHandler.startHandler , args=( ) )
                thr.setDaemon(True)
                thr.start()
                thr.join()

                skt.close()
                print("Socket {0} handling is done............".format(addr))



if __name__ == '__main__':
    ws = WebServer()
    ws.start()

添加配置文件

1.新建文件ServerContent

class ServerContent:
    ip = '127.0.0.1'
    port = 9999

    head_protocal = "HTTP/1.1 "
    head_code_200 = "200 "
    head_status_OK = "OK"

    head_content_length = "Content-Length: "
    head_content_type = "Content-Type: "
    content_type_html = "text/html"

    blank_line = ""
